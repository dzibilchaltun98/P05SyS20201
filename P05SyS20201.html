Codigo del trapecio

function [intAprox] = trapecio(f_t, cantTrap, minT, maxT)
    f0 = f_t(minT);
    fn = f_t(maxT);
    h = (maxT - minT)/cantTrap;

    fs = 0;
    for t = 1:cantTrap-1
        xn(t) = minT + h*t;
        fs = fs + f_t(xn(t));
    end

    intAprox = (h/2)*(f0 + 2*fs + fn);
end
